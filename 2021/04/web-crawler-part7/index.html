Web Crawler Part7: 使用文档型数据库开发爬虫项目

Web Crawler Part7

之前的文章中由索引问题出发提及了非InnoDB引擎的数据库，在这篇文章中，我们尝试使用文档型数据库MongoDB开发这个项目

我们尝试两种方法进行开发

  1. 从MySQL中迁移爬取的数据
  2. 直接将爬取的数据写入MongoDB

Section 1: 配置MongoDB与Node.js相应模块

(Remark: 博主使用的是MacOS系统进行开发，Windows系统上的使用应该大同小异)

在一切开始之前，我们需要先下载一个MongoDB

MongoDB Server: https://www.mongodb.com/try/download/community （也可以选择使用命令行）

MongoDB Compass: https://www.mongodb.com/try/download/shell

我们先在本地安装MongoDB和MongoDB Compass（后者是图形化界面，性质类似于MySQL Workbench）

安装流程如下：

# 切换到 /usr/local 目录之下
cd /usr/local
# 下载MongoDB的MacOS版本压缩包
sudo curl -O https://fastdl.mongodb.org/osx/mongodb-osx-ssl-x86_64-4.0.9.tgz
# 解压文件
sudo tar -zxvf mongodb-osx-ssl-x86_64-4.0.9.tgz
# 将其重命名为 Mongodb 
sudo mv mongodb-osx-x86_64-4.0.9/ mongodb

随后配置环境变量和路径

export PATH=/usr/local/mongodb/bin:$PATH
sudo mkdir -p /usr/local/var/mongodb
sudo mkdir -p /usr/local/var/log/mongodb

Remark:可能需要对该路径的操作权限，如下

sudo chown myname /usr/local/var/mongodb
sudo chown myname /usr/local/var/log/mongodb

配置好环境后，我们在命令行工具中启动MongoDB

使用npm安装mongoDB模组

npm install mongo

然后我们编写一个简单的Js程序，操作我们的数据库

var MongoClient = require('mongodb').MongoClient;
var url = 'mongodb://localhost:27017/';
MongoClient.connect(url, function (err, db) {
    if (err) throw err;
    console.log('Database has already been initialized');
    var dbase = db.db("crawler");
    dbase.createCollection('MyFetch', function (err, res) {
        if (err) throw err;
        console.log("create MyFetch!");
        db.close();
    });
});

我们建立一个爬虫数据库，再初始化一张表MyFetch用于存储我们的爬虫数据

Section 2: 从MySQL迁移数据

方便起见，在使用Node.js直接将爬取的数据写入MongoDB前，我们先将已经爬取的新闻数据迁移到MongoDB中，一方面是不想丢掉稍微老一些的新闻数据，另一方面检查一下是否出现乱码或配置错误等问题

随后，我们将之前存储在MySQL数据库中的新闻导出成json格式，并且通过MongoDB Compass导入我们在crawler数据库中的MyFetch表之中

我们导出crawler_data.json，将其导入MongoDB

迁移完毕，我们可以通过MongoDB Compass来查看效果，内容与关系型数据库中存储的新闻一致。

Section 3: 将爬取的数据写入MongoDB

把爬取数据写入MongoDB的方式和写入关系型数据库没有太大的差别，主要是需要把一些Node.js调用Sql的语法替换为调用mongoDB的语法，如下

var MongoClient = require('mongodb').MongoClient;
var url = "mongodb://localhost:27017/"; //默认url

function newsGet(myURL, format) { 
    request(myURL, function(err, res, body) { 
        try {
            var html_news = myIconv.decode(body, format.myEncoding); //用iconv-module进行转码
            // console.log(html_news);
            //准备用cheerio解析html_news
            var $ = myCheerio.load(html_news, { decodeEntities: true });
            myhtml = html_news;
        } catch (e) {    console.log('读新闻页面并转码出错：' + e);};

        console.log("转码读取成功:" + myURL);
        var fetch = {};
        fetch.title = "";
        fetch.content = "";
        fetch.publish_date = (new Date()).toFormat("YYYY-MM-DD");
        //fetch.html = myhtml;
        fetch.url = myURL;
        fetch.source_name = format.source_name;
        fetch.source_encoding = format.myEncoding; 
        fetch.crawltime = new Date();

        if (fetch.keywords_format == "") fetch.keywords = format.source_name; // eval(keywords_format);  //没有关键词就用sourcename
        else fetch.keywords = eval(format.keywords_format);

        if (format.title_format == "") fetch.title = ""
        else fetch.title = eval(format.title_format); //Title

        if (format.author_format == "") fetch.author = format.source_name; //eval(author_format);  //Author
        else fetch.author = eval(format.author_format);
        console.log('author: ' + fetch.author);

        if (format.date_format != "") fetch.publish_date = eval(format.date_format); //Date
        console.log('date: ' + fetch.publish_date);

        if (fetch.publish_date == undefined);
        else {
            fetch.publish_date = format.regExp.exec(fetch.publish_date)[0];
            fetch.publish_date = fetch.publish_date.replace('年', '-')
            fetch.publish_date = fetch.publish_date.replace('月', '-')
            fetch.publish_date = fetch.publish_date.replace('日', '')
            fetch.publish_date = new Date(fetch.publish_date).toFormat("YYYY-MM-DD");
        }

        if (format.content_format == "") fetch.content = "";
        else fetch.content = eval(format.content_format).replace("\r\n" + fetch.author, ""); //Content

        if (format.source_format == "") fetch.source = fetch.source_name;
        else fetch.source = eval(format.source_format).replace("\r\n", ""); //Source

        if (format.desc_format == "") fetch.desc = fetch.title;
        else  {
            fetch.desc = eval(format.desc_format); //Summary
            if (fetch.desc == undefined) fetch.desc = "";
            else fetch.desc.replace("\r\n", "");
        }   

        // 注释掉原先MySQL数据库所使用的语句，替换为一句极其简单的mongoDB插入语句
       	// var fetchAddSql = 'INSERT INTO fetches(url,source_name,source_encoding,title,' +
        // 'keywords,author,publish_date,crawltime,content, description) 				VALUES(?,?,?,?,?,?,?,?,?,?)';
        // var fetchAddSql_Params = [fetch.url, fetch.source_name, fetch.source_encoding,
        // fetch.title, fetch.keywords, fetch.author, fetch.publish_date,
        // fetch.crawltime.toFormat("YYYY-MM-DD HH24:MI:SS"), fetch.content, fetch.desc];
          MongoClient.connect(url, function(err, db) {
          if (err) throw err;
          var mongo_db = db.db("crawler");
          try{ //开始文档插入
              mongo_db.collection("myFetch").insertOne(fetch, function(err, res) 
              {
                  if (err){
                    console.log('url error:' + err);
                  } 
                  console.log("document successfully inserted");
                  db.close();
              });
          } catch (e) { console.log('url error:' + e) };
          });
        // 对比db.COLLECTION_NAME.insert(document)，文档型数据库的插入非常简单!
        }); 
    });
}

到此为止，我们已经可以使用MongoDB进行这个爬虫项目的开发了，下一篇文章将会对爬取的数据进行分析，做一些fancy的东西

继续实现分页功能，确定如何索引到需要的数据

此处使用了Lodash 其通过降低 array、number、objects、string 等等的使用复杂度从而让 JavaScript变得清晰，我们借助它做数据遍历再生成页码列表

先要获取数据，代码如下

getPagedData = () => {
    const {
      pageSize,
      currentPage,
      sortColumn,
      selectedGenre,
      movies: allMovies,
    } = this.state;
    const filtered =
      selectedGenre && selectedGenre._id
        ? allMovies.filter((movie)=>this.inSort(movie, selectedGenre))
        : allMovies; //逻辑需要封装
    const sorted = _.orderBy(filtered, [sortColumn.path], [sortColumn.order]);
    const movies = paginate(sorted, currentPage, pageSize);
	// movie作为数据对象返回！
    return { totalCount: filtered.length, data: movies };
  };

import React, { Component } from "react";
import _ from "lodash";
const Pagination = (props) => {
  const { itemsCount, pageSize } = props; //计算得出有多少页码，这个计算结果是一个浮点数
  const pagesCount =  itemsCount / pageSize ; //需要利用lodash包中的方法，可以生成一个页码的列表
  const pages = _.range(1, pagesCount + 1); //如果当前的总信息条数小于单页信息条数，我们就不用渲染翻页按钮了
  if (pagesCount <= 1) return null;
  return (
    <nav>
      <ul className="pagination">
          {
        {pages.map((page) => ( // 利用map函数，每个标生成数字标签
          <li key={page} className="page-item"> 
            <a className="page-link">{page}</a>
          </li>
        ))}
      </ul>
    </nav>
  );
};

  handleSubmit = (e) => {
